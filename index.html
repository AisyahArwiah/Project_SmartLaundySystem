AND password = ?"; try (Connection conn = DBConnection.getConnection()) { PreparedStatement ps = conn.prepareStatement(query); ps.setString(1, email); ps.setString(2, password); try (ResultSet rs = ps.executeQuery()) { if (rs.next()) { User user = new User(); user.setUserID(rs.getInt("userID")); user.setName(rs.getString("name")); user.setEmail(rs.getString("email")); return user; } } } return null; } public int getUserPoints(int userID) throws SQLException { String query = "SELECT points FROM loyalty_accounts WHERE userID = ?"; try (Connection conn = DBConnection.getConnection()) { PreparedStatement ps = conn.prepareStatement(query); ps.setInt(1, userID); try (ResultSet rs = ps.executeQuery()) { if (rs.next()) return rs.getInt("points"); } } return 0; } }
